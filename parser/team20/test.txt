CREATE TABLE Usuarios (
    id INTEGER PRIMARY KEY ,
    nombre VARCHAR (8) DEFAULT 1234 UNIQUE ,
    clave VARCHAR (10) UNIQUE NOT NULL DEFAULT 4,
    edad INTEGER,
    CONSTRAINT f_edad CHECK (edad>18)
);

CREATE TABLE Casas (
    id INTEGER ,
    grupo CHAR CONSTRAINT f_grupo UNIQUE,
    clave INTEGER CONSTRAINT f_calve CHECK (clave>10) DEFAULT 0,
    fecha TIMESTAMP NULL CHECK (fecha>20),
    PRIMARY KEY (id)
)INHERITS(Usuarios);
